a20

Implemented By
x86-486sx, x86-hammer, x86-p2, x86-p4
Description
This interface is used between the A20 line handling device (typically the keyboard controller) and the x86 processor. 
typedef struct {
        void (*set_a20_line)(conf_object_t *NOTNULL, int);
        int (*get_a20_line)(conf_object_t *NOTNULL);
} a20_interface_t;
