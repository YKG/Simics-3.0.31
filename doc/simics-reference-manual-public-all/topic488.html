apic_cpu

Implemented By
apic
Description
Interface to read and write the task priority register. Only used on x86-64 where the task priority register is both a cpu register (CR8) and a register in the local apic. 

struct apic_cpu_interface {
        uint64 (*tpr_r)(conf_object_t *NOTNULL obj);
        void (*tpr_w)(conf_object_t *NOTNULL obj, uint64 tpr);
        void (*init)(conf_object_t *NOTNULL obj);
};
